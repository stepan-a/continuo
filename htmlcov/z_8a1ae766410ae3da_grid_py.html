1"""Time grids for the perfect-foresight solve. 

2 

3A grid over ``[t₀, t_N]`` is just its ``N + 1`` node positions; the 

4per-interval steps, the interval midpoints (which Crank–Nicolson evaluates 

5at), and the interval count are derived from them. The nodes need not be 

6equally spaced — :func:`uniform_grid` builds the equal-step default, and 

7:func:`mesh_from_points` wraps an arbitrary increasing node sequence (for a 

8graded or refined mesh). Multi-segment grids (with reveal-time breakpoints) 

9are built by the orchestrator on top of these. 

10""" 

11 

12from __future__ import annotations 

13 

14from dataclasses import dataclass 

15 

16import numpy as np 

17 

18from continuo.solve.errors import SolveError 

19 

20__all__ = ["Grid", "uniform_grid", "mesh_from_points", "aligned_grid"] 

21 

22 

23@dataclass(eq=False) 

24class Grid: 

25 """A time grid, defined by its strictly increasing node positions. 

26 

27 ``steps`` (the ``N`` per-interval widths), ``midpoints`` (the ``N`` 

28 interval midpoints) and ``intervals`` (``N``) are derived from 

29 ``points`` (``N + 1`` nodes), so a non-uniform mesh is just a ``points`` 

30 array with unequal gaps. 

31 """ 

32 

33 points: np.ndarray 

34 

35 @property 

36 def steps(self) -> np.ndarray: 

37 """The ``N`` per-interval widths ``tᵢ₊₁ − tᵢ``.""" 

38 return np.diff(self.points) 

39 

40 @property 

41 def midpoints(self) -> np.ndarray: 

42 """The ``N`` interval midpoints.""" 

43 return 0.5 * (self.points[:-1] + self.points[1:]) 

44 

45 @property 

46 def intervals(self) -> int: 

47 return len(self.points) - 1 

48 

49 @property 

50 def dt(self) -> float: 

51 """The single step of a uniform grid; raises for a non-uniform mesh.""" 

52 steps = self.steps 

53 if not np.allclose(steps, steps[0]): 

54 raise ValueError("non-uniform grid has no single dt; use .steps") 

55 return float(steps[0]) 

56 

57 

58def uniform_grid(horizon: float, intervals: int, start: float = 0.0) -> Grid: 

59 """Build a uniform grid over ``[start, start + horizon]`` with ``intervals`` intervals. 

60 

61 ``start`` offsets the grid so a simulation segment beginning at a 

62 reveal time uses absolute times (the shock paths and ``t`` are in 

63 absolute time). 

64 """ 

65 if horizon <= 0: 

66 raise SolveError("simulation horizon T must be positive") 

67 if intervals < 1: 

68 raise SolveError("number of grid intervals N must be at least 1") 

69 return Grid(points=float(start) + np.linspace(0.0, float(horizon), intervals + 1)) 

70 

71 

72def mesh_from_points(points: np.ndarray) -> Grid: 

73 """Wrap an arbitrary strictly-increasing node sequence as a :class:`Grid`.""" 

74 pts = np.asarray(points, dtype=float) 

75 if pts.ndim != 1 or len(pts) < 2: 

76 raise SolveError("a grid needs at least two points") 

77 if np.any(np.diff(pts) <= 0): 

78 raise SolveError("grid points must be strictly increasing") 

79 return Grid(points=pts) 

80 

81 

82def aligned_grid(start: float, horizon: float, intervals: int, mark: float) -> tuple[Grid, int]: 

83 """A mesh on ``[start, start + horizon]`` (``intervals`` intervals) with a node at ``mark``. 

84 

85 Returns the grid and the index of the ``mark`` node — the next reveal time 

86 of a segment, or the terminal time ``T`` for the last one — so it falls on 

87 a node instead of being snapped to the nearest one. When ``mark`` is the far 

88 end (or ``intervals < 2``) the mesh is uniform. 

89 """ 

90 far = start + horizon 

91 if intervals < 2 or mark >= far: 

92 return uniform_grid(horizon, intervals, start=start), intervals 

93 m = min(max(int(round(intervals * (mark - start) / horizon)), 1), intervals - 1) 

94 left = np.linspace(start, mark, m + 1) 

95 right = np.linspace(mark, far, intervals - m + 1) 

96 return mesh_from_points(np.concatenate([left, right[1:]])), m
