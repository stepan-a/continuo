1"""Assembling the model residual and its Jacobians. 

2 

3The model is the DAE ``F(ẋ, x, e, θ, t) = 0``. This module stacks the 

4per-equation residuals — ``lhs - rhs`` for an explicit equation, ``expr`` 

5for a bare ``expr;`` (which means ``expr == 0``) — into the vector ``F`` 

6and derives its Jacobians by CasADi automatic differentiation. 

7 

8Vector orderings: 

9 

10- ``x`` : endogenous variables in ``model.endogenous`` order; 

11- ``ẋ`` : time derivatives, states then jumps (the variables that have one); 

12- ``e`` : exogenous processes in ``model.exogenous`` order; 

13- ``θ`` : parameters in ``model.parameters`` order; 

14- ``t`` : the scalar reserved time. 

15 

16The result holds CasADi ``Function`` objects for ``F`` and for ``∂F/∂x`` 

17and ``∂F/∂ẋ`` (the blocks the collocation Newton needs). C codegen and 

18JIT compilation of these are a later step. 

19""" 

20 

21from __future__ import annotations 

22 

23from dataclasses import dataclass 

24 

25import casadi as ca 

26 

27from continuo.codegen.translate import SymbolTable, build_symbols, translate 

28from continuo.ir.model import Model 

29 

30__all__ = ["Residual", "build_residual"] 

31 

32 

33@dataclass 

34class Residual: 

35 """The symbolic residual ``F`` of a model and its Jacobians, as CasADi. 

36 

37 ``function`` maps ``(ẋ, x, e, θ, t)`` to the residual vector; 

38 ``jacobian_x`` and ``jacobian_xdot`` map the same inputs to ``∂F/∂x`` 

39 and ``∂F/∂ẋ`` respectively. 

40 """ 

41 

42 symbols: SymbolTable 

43 expression: ca.SX 

44 function: ca.Function 

45 jacobian_x: ca.Function 

46 jacobian_xdot: ca.Function 

47 

48 

49def build_residual(model: Model) -> Residual: 

50 """Build the residual ``F`` and its Jacobians for ``model``.""" 

51 table = build_symbols(model) 

52 

53 residuals = [_residual(eq, table) for eq in model.equations] 

54 F = _column(residuals) 

55 

56 x = _column([table.symbols[name] for name in model.endogenous]) 

57 xdot = _column([table.derivatives[name] for name in (*model.states, *model.jumps)]) 

58 e = _column([table.symbols[name] for name in model.exogenous]) 

59 theta = _column([table.symbols[name] for name in model.parameters]) 

60 t = table.symbols["t"] 

61 

62 inputs = [xdot, x, e, theta, t] 

63 names = ["xdot", "x", "e", "theta", "t"] 

64 function = ca.Function("F", inputs, [F], names, ["F"]) 

65 jacobian_x = ca.Function("jac_x", inputs, [ca.jacobian(F, x)], names, ["jac_x"]) 

66 jacobian_xdot = ca.Function("jac_xdot", inputs, [ca.jacobian(F, xdot)], names, ["jac_xdot"]) 

67 

68 return Residual(table, F, function, jacobian_x, jacobian_xdot) 

69 

70 

71def _residual(eq, table: SymbolTable) -> ca.SX: 

72 if eq.lhs is None: # bare `expr;` means expr == 0 

73 return translate(eq.rhs, table) 

74 return translate(eq.lhs, table) - translate(eq.rhs, table) 

75 

76 

77def _column(symbols: list[ca.SX]) -> ca.SX: 

78 return ca.vertcat(*symbols) if symbols else ca.SX.zeros(0, 1)
